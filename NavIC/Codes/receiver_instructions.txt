cd ~/FWC/NavIC/Codes/Receiver/bin

BladeRF Configuration
---------------------
bladeRF-cli -i <<EOF
set frequency rx 1176.45e6
set frequency tx 47e6
set agc off
set samplerate 2.048e6
set bandwidth 3.0e6
set gain rx 60
set gain tx 0
set clock sel onboard
set biastee rx off
EOF
